max_output_value= 499
neighborhood_size= 299
subtract_from_mean= 0

image_binarized= cv2.adaptiveThreshold(image_grey,
                                      max_output_value,
                                      cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
                                      cv2.THRESH_BINARY,
                                      neighborhood_size,
                                      subtract_from_mean)
